# AY-3-8910 Music Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Play Pator's `Spectrumizer.pt3` AY chiptune on the title screen and continuously in-game on any machine with an AY chip, auto-detected at runtime, while keeping the existing 1-bit beeper SFX unchanged; silent and unchanged on the beeper-only TC2048.

**Architecture:** A new target-only `music` module owns all AY I/O (mirrors how `scld.c` owns `0xFF`). It probes for an AY on two port schemes (128K `0xFFFD/0xBFFD`, TS2068 `0x00F5/0x00F6`), latches the one that answers, and drives a vendored ProTracker-3 player once per frame from the main loop (no interrupt-mode change). The beeper SFX layer is untouched and mixes over the music in hardware.

**Tech Stack:** C99 + Z80 asm, z88dk `zcc +zx -SO3 -clib=sdcc_iy`, z80asm directives (`SECTION`/`PUBLIC`/`EXTERN`/`BINARY`), ZEsarUX (TC2048 + 128K/TS2068 models) for audio, `z88dk-ticks` for the perf gate. Host tests via `cc` (`test/run.sh`) — unaffected (music is a target-only no-op on the host).

## Global Constraints

- **No float, no `malloc`, no recursion.** Integer math, fixed arrays, precomputed tables only.
- **Never return a struct by value — pass via out-pointer** (sdcc Z80 SIGSEGVs otherwise).
- **Pass to asm via globals, parameterless routines** (foundation spec §15.5), like `sfx_hp`/`sfx_dur` + `sfx_blip`.
- **Asm never touches IY** (sdcc_iy frame pointer) and never sets port `0xFF` bits 6-7.
- **Only `scld.c` knows `0xFF`/screen addresses; only the `music` module knows AY ports; only `sfx.asm` knows the beeper.**
- **Frame budget = 69,888 T @ 50 Hz (~55k usable).** Always profile with `z88dk-ticks` before claiming a perf change.
- **AY port schemes:** 128K/+2/+3 (and 48K+AY) = select `0xFFFD`, write `0xBFFD`, read `0xFFFD`. TS2068/TC2068 = select `0x00F5`, write `0x00F6`, read `0x00F6`.
- **Build runs `./build.sh` → `build/game.tap`.** Verification cycle for this module = build + ZEsarUX (audio) + `z88dk-ticks` (budget); there are no host unit tests for hardware code.
- **ORG `0x8000`; keep player code/data/work-RAM out of `0x4000–0x7FFF`** (both screens + back buffer live there).

- [ ] **Step 1: Write `src/music_ay.asm` — detection + register write**

```asm
; music_ay.asm -- AY-3-8910 presence/port detection and register output.
; Probes two port schemes, latches the one that answers, exposes a C-callable
; ay_put (reg in _ay_reg, value in _ay_val). Parameterless/globals (spec 15.5);
; never touches IY; never writes port 0xFF.
        SECTION code_user

        PUBLIC  _ay_detect      ; u8  ay_detect(void)  -> 1 if AY found, else 0
        PUBLIC  _ay_put         ; void ay_put(void): AY[_ay_reg] = _ay_val
        PUBLIC  _ay_sel         ; u16 latched select port (player + ay_put use)
        PUBLIC  _ay_dat         ; u16 latched data   port
        EXTERN  _ay_reg         ; u8  (set by C before ay_put)
        EXTERN  _ay_val         ; u8  (set by C before ay_put)

; candidate (sel, dat, read) triples, sel==0 terminates.
ay_pairs:
        defw 0xFFFD, 0xBFFD, 0xFFFD     ; ZX 128/+2/+3 and 48K+AY
        defw 0x00F5, 0x00F6, 0x00F6     ; TS2068 / TC2068
        defw 0x0000, 0x0000, 0x0000

_ay_sel: defw 0
_ay_dat: defw 0
ay_rd:   defw 0

_ay_detect:
        ld   hl,ay_pairs
det_next:
        ld   e,(hl)
        inc  hl
        ld   d,(hl)
        inc  hl
        ld   a,e
        or   d
        jr   z,det_fail         ; sel==0 -> table end, no AY
        ld   (_ay_sel),de
        ld   c,(hl)
        inc  hl
        ld   b,(hl)
        inc  hl
        ld   (_ay_dat),bc
        ld   c,(hl)
        inc  hl
        ld   b,(hl)
        inc  hl
        ld   (ay_rd),bc
        push hl
        call ay_probe           ; A=1 if both patterns echo
        pop  hl
        or   a
        jr   nz,det_ok
        jr   det_next
det_ok:
        ld   a,1
        ret
det_fail:
        xor  a
        ret

; ay_probe -- with the latched ports, R0 := 0x55 read-back, then 0xAA. A=1 both.
ay_probe:
        ld   b,0
        ld   c,0x55
        call set_rc
        ld   b,0
        call get_r
        cp   0x55
        jr   nz,probe_no
        ld   b,0
        ld   c,0xAA
        call set_rc
        ld   b,0
        call get_r
        cp   0xAA
        jr   nz,probe_no
        ld   b,0
        ld   c,0
        call set_rc             ; restore R0=0
        ld   a,1
        ret
probe_no:
        xor  a
        ret

; set_rc -- select reg B, write value C (ay_sel/ay_dat). Clobbers A,BC.
set_rc:
        ld   a,b
        push bc
        ld   bc,(_ay_sel)
        out  (c),a
        pop  bc
        ld   a,c
        ld   bc,(_ay_dat)
        out  (c),a
        ret

; get_r -- select reg B, A := value (ay_rd). Clobbers A,BC.
get_r:
        ld   a,b
        ld   bc,(_ay_sel)
        out  (c),a
        ld   bc,(ay_rd)
        in   a,(c)
        ret

; _ay_put -- AY[_ay_reg] := _ay_val (C-callable via globals).
_ay_put:
        ld   a,(_ay_reg)
        ld   bc,(_ay_sel)
        out  (c),a
        ld   a,(_ay_val)
        ld   bc,(_ay_dat)
        out  (c),a
        ret
```

- [ ] **Step 2: Write `include/music.h`**

```c
/*
 * music.h -- AY chiptune music (target-only). Owns all AY-port access, exactly
 * as scld.c owns port 0xFF. On a beeper-only machine (e.g. TC2048) music_init
 * returns 0 and every other entry is a cheap no-op.
 */
#ifndef MUSIC_H
#define MUSIC_H

#include "types.h"

/* Probe for an AY, latch its port pair, start the tune. Returns 1 if found. */
u8   music_init(void);

/* Advance the player one 50 Hz frame. Call once per HALT. No-op if no AY. */
void music_tick(void);

/* Silence all AY channels (defined for completeness; unused in this slice). */
void music_stop(void);

#endif /* MUSIC_H */
```

- [ ] **Step 3: Write `src/music.c` (tone mode for now)**

```c
/*
 * music.c -- AY music module (see music.h). Target-only; host build is a no-op
 * so host unit tests never link hardware. THIS revision plays a swept test tone
 * to prove detection + port routing; Task 3 swaps the body for the PT3 player.
 */
#include "music.h"

#ifdef __SDCC

/* asm side (music_ay.asm) */
extern u8  ay_detect(void);
extern void ay_put(void);
u8  ay_reg;     /* set before ay_put() */
u8  ay_val;

static u8 music_on;     /* 1 once an AY is detected */
static u8 tone_lo;      /* swept tone period low byte */

static void put(u8 r, u8 v) { ay_reg = r; ay_val = v; ay_put(); }

u8 music_init(void)
{
    music_on = ay_detect();
    if (!music_on) {
        return 0;
    }
    /* Steady tone on channel A: mixer tone-A enabled (bit0=0), noise off;
     * amplitude mid; a starting period. (R7 bits are active-low enables.) */
    put(7u, 0x3Eu);     /* 0b00111110: tone A on, others off */
    put(8u, 0x0Cu);     /* channel A amplitude */
    put(0u, 0x80u);     /* tone A fine */
    put(1u, 0x01u);     /* tone A coarse */
    tone_lo = 0x80u;
    return 1;
}

void music_tick(void)
{
    if (!music_on) {
        return;
    }
    tone_lo = (u8)(tone_lo + 1u);   /* audible sweep proves the per-frame tick */
    put(0u, tone_lo);
}

void music_stop(void)
{
    if (!music_on) {
        return;
    }
    put(8u, 0u);   /* amplitude 0 */
    put(9u, 0u);
    put(10u, 0u);
}

#else /* host build -- pure no-op */
u8   music_init(void) { return 0; }
void music_tick(void) {}
void music_stop(void) {}
#endif
```

- [ ] **Step 2: Port-adapt the player's AY output**

Find the player's AY-output stage — the instructions that load the AY ports immediately and `out (c),a` (typically `ld bc,#fffd : out (c),a` to select and `ld bc,#bffd : out (c),a` to write), or an unrolled 14-register dump. Replace the **immediate** port loads with **indirect** loads from the latched ports, so the same player serves both schemes:

```asm
; was:  ld bc,#fffd  /  out (c),a      (select reg)
;       ld bc,#bffd  /  out (c),a      (write value)
; now:  ld bc,(_ay_sel) / out (c),a
;       ld bc,(_ay_dat) / out (c),a
```

If the player dumps a 14-byte computed register buffer (commonly labelled `AYREGS`), replace that dump with this routine (call it where the player wrote the AY), preserving the PT3 "don't retrigger envelope" rule (skip R13 when it is `0xFF`):

```asm
; ay_out_regs -- HL -> 14-byte AYREGS; write all via the latched ports.
ay_out_regs:
        ld   d,0                ; D = register index 0..13
aor_lp:
        ld   a,d
        cp   13
        jr   nz,aor_w
        ld   a,(hl)
        cp   0xFF
        jr   z,aor_skip         ; R13==0xFF -> no envelope retrigger
        ld   a,d
aor_w:
        ld   bc,(_ay_sel)
        out  (c),a              ; select reg D
        ld   a,(hl)
        ld   bc,(_ay_dat)
        out  (c),a              ; write value
aor_skip:
        inc  hl
        inc  d
        ld   a,d
        cp   14
        jr   nz,aor_lp
        ret
```
